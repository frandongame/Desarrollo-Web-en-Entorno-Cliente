function rot13(str) {
  let res = '';
  for (let i = 0; i < str.length; i++) {
    let c = str.charCodeAt(i);
    if (c >= 65 && c <= 90) {
      res += String.fromCharCode(((c - 65 + 13) % 26) + 65);
    } else {
      res += str[i];
    }
  }
  return res;
}

rot13("SERR PBQR PNZC");
